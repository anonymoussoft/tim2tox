# Option 1: Using FetchContent (Recommended for header-only libraries)
# Add this to tim2tox/ffi/CMakeLists.txt before add_library()

include(FetchContent)
FetchContent_Declare(json
    URL https://github.com/nlohmann/json/releases/download/v3.12.0/json.tar.xz
    # Or use a specific commit for stability:
    # GIT_REPOSITORY https://github.com/nlohmann/json.git
    # GIT_TAG v3.12.0
)
set(JSON_Install OFF CACHE INTERNAL "")
set(JSON_BuildTests OFF CACHE INTERNAL "")
FetchContent_MakeAvailable(json)

# Then link it to tim2tox_ffi:
target_link_libraries(tim2tox_ffi PRIVATE nlohmann_json::nlohmann_json)
